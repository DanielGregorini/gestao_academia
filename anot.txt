version: '3.8'

services:
  mysql:
    image: mysql:8.0
    environment:
      MYSQL_ROOT_PASSWORD: admin
      MYSQL_DATABASE: db_academia
    ports:
      - "3306:3306"
    volumes:
      - mysql-data:/var/lib/mysql
    restart: always

  academia_api:
    build:
      context: ./backend/academia_api
    environment:
      - ConnectionStrings__DefaultConnection=Server=mysql;Database=db_academia;User=root;Password=admin;
    depends_on:
      - mysql
    ports:
      - "5777:5777"
    volumes:
      - ./backend/academia_api:/app
      - /app/bin # Evita problemas com compilação e binários sobrescritos
    restart: always
    
  academia_frontend:
    build:
      context: ./frontend/academia
    environment:
      - NODE_ENV=development
      - NEXT_PUBLIC_API_URL=http://host.docker.internal:5777 # Acessa o backend na máquina host
    depends_on:
      - academia_api
    ports:
      - "3000:3000"
    volumes:
      - ./frontend/academia:/app
      - /app/node_modules
    command: yarn dev
